/* -------------------------------------------------
   Flipkart‑Best‑Product 3‑D Demo
   -------------------------------------------------
   • Uses a placeholder product (hard‑coded) because
     Flipkart’s public API is restricted.
   • Replace the placeholder data with a real API call
     if you have credentials.

   • The 3‑D object is a simple rotating box whose
     texture is the product image.
   ------------------------------------------------- */

const PRODUCT = {
  name: "Flipkart Smart‑Watch Series 5",
  price: "₹4,999",
  // Open‑source placeholder image – swap with real URL if you have one
  image: "https://images.unsplash.com/photo-1615190175065-5d5ce5b7c1bc?w=400&fit=crop"
};

// -------------------------------------------------
// 1️⃣ Populate the HTML overlay with product data
// -------------------------------------------------
document.getElementById('product-name').textContent = PRODUCT.name;
document.getElementById('product-price').textContent = `Price: ${PRODUCT.price}`;

// -------------------------------------------------
// 2️⃣ Set up Three.js scene
// -------------------------------------------------
const canvas = document.getElementById('canvas');
const renderer = new THREE.WebGLRenderer({ canvas, antialias: true });
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.setPixelRatio(window.devicePixelRatio);

const scene = new THREE.Scene();
scene.background = new THREE.Color(0xf0f0f0);

const camera = new THREE.PerspectiveCamera(
  45,
  window.innerWidth / window.innerHeight,
  0.1,
  1000
);
camera.position.set(0, 1.5, 3);

// OrbitControls – allows mouse drag rotate + zoom
const controls = new THREE.OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;

// -------------------------------------------------
// 3️⃣ Add a light source
// -------------------------------------------------
const ambient = new THREE.AmbientLight(0xffffff, 0.7);
scene.add(ambient);

const dirLight = new THREE.DirectionalLight(0xffffff, 0.5);
dirLight.position.set(5, 10, 7);
scene.add(dirLight);

// -------------------------------------------------
// 4️⃣ Create a box geometry and map the product image
// -------------------------------------------------
const geometry = new THREE.BoxGeometry(1, 1, 1);
const loader = new THREE.TextureLoader();
loader.crossOrigin = ""; // allow CORS for external image

loader.load(
  PRODUCT.image,
  texture => {
    const material = new THREE.MeshStandardMaterial({ map: texture });
    const cube = new THREE.Mesh(geometry, material);
    scene.add(cube);
    animate();

    // Simple spin animation (still works with OrbitControls)
    function animate() {
      requestAnimationFrame(animate);
      cube.rotation.y += 0.005;
      controls.update();
      renderer.render(scene, camera);
    }
  },
  undefined,
  err => {
    console.error('Failed to load product image', err);
  }
);

// -------------------------------------------------
// 5️⃣ Handle window resize
// -------------------------------------------------
window.addEventListener('resize', () => {
  const w = window.innerWidth;
  const h = window.innerHeight;
  renderer.setSize(w, h);
  camera.aspect = w / h;
  camera.updateProjectionMatrix();
});
