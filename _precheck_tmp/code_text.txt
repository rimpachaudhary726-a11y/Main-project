import sys
import datetime
import urllib.request
import json
from typing import List, Tuple

API_BASE = "https://api.exchangerate.host"


def fetch_json(url: str) -> dict:
    """Retrieve JSON data from a URL, raising RuntimeError on failure."""
    try:
        with urllib.request.urlopen(url, timeout=10) as resp:
            if resp.status != 200:
                raise RuntimeError(f"HTTP {resp.status} for {url}")
            data = resp.read().decode()
            return json.loads(data)
    except Exception as e:
        raise RuntimeError(f"Failed to fetch {url}: {e}")


def get_rate(base: str, target: str, date: str = None) -> float:
    """
    Return the exchange rate for base→target.
    If `date` is None the latest rate is requested, otherwise historical rate for that date.
    """
    if date:
        url = f"{API_BASE}/{date}?base={base}&symbols={target}"
    else:
        url = f"{API_BASE}/latest?base={base}&symbols={target}"
    data = fetch_json(url)
    try:
        return float(data["rates"][target])
    except (KeyError, TypeError, ValueError):
        raise RuntimeError(f"Rate not found for {base}/{target} (date={date})")


def parse_args(args: List[str]) -> List[Tuple[str, str]]:
    """
    Convert command‑line arguments into a list of (base, target) tuples.
    Supports two styles:
        USD EUR GBP JPY            -> [(USD, EUR), (GBP, JPY)]
        USD/EUR GBP/JPY            -> [(USD, EUR), (GBP, JPY)]
    """
    pairs = []
    i = 0
    while i < len(args):
        token = args[i].strip()
        if "/" in token:
            parts = token.split("/", 1)
            if len(parts) != 2 or not all(parts):
                raise ValueError(f"Invalid pair specifier: '{token}'")
            base, target = parts[0].upper(), parts[1].upper()
            pairs.append((base, target))
            i += 1
        else:
            if i + 1 >= len(args):
                raise ValueError(f"Missing target currency for base '{token}'")
            base = token.upper()
            target = args[i + 1].strip().upper()
            pairs.append((base, target))
            i += 2
    return pairs


def format_change(pct: float) -> str:
    """Return a nicely formatted percentage change string with a sign."""
    sign = "+" if pct >= 0 else ""
    return f"{sign}{pct:.2f}%"


def main() -> None:
    if len(sys.argv) < 2:
        print(
            "Usage:\n"
            "  python exchange_tool.py BASE TARGET [BASE TARGET ...]\n"
            "  or\n"
            "  python exchange_tool.py USD/EUR GBP/JPY ..."
        )
        sys.exit(1)

    try:
        pairs = parse_args(sys.argv[1:])
    except ValueError as ve:
        print(f"Argument error: {ve}")
        sys.exit(2)

    # Date a week ago (most recent date for which the API provides data)
    week_ago = (datetime.datetime.utcnow() - datetime.timedelta(days=7)).date()
    week_ago_str = week_ago.isoformat()

    # Header
    header = f"{'Pair':<12} {'Rate':>12} {'7‑day %':>10}"
    print(header)
    print("-" * len(header))

    for base, target in pairs:
        try:
            latest = get_rate(base, target)
            past = get_rate(base, target, date=week_ago_str)
            pct_change = ((latest - past) / past) * 100 if past != 0 else 0.0
            pair_str = f"{base}/{target}"
            rate_str = f"{latest:.6f}"
            change_str = format_change(pct_change)
            print(f"{pair_str:<12} {rate_str:>12} {change_str:>10}")
        except RuntimeError as err:
            print(f"{base}/{target:<12} Error: {err}")

if __name__ == "__main__":
    main()
